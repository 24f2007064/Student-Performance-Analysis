```python
import os

from sklearn.cluster import KMeans

from sklearn.model_selection import cross_val_score
from sklearn.metrics import mean_absolute_error, r2_score
import numpy as np

# ── STEP 1: Build wide format cleanly ──────────────────────────────
df_wide = df.pivot_table(
    index='Student_ID',
    columns='Semester',
    values='SGPA'
)
df_wide.columns = [f'S{int(c)}' for c in df_wide.columns]
df_wide = df_wide.reset_index()

# Attach Performance_Type and QB_Era back
static = df.drop_duplicates('Student_ID')[['Student_ID', 'Batch',

df_wide = df_wide.merge(static, on='Student_ID')

print("Wide shape:", df_wide.shape)

```

```python
# ── STEP 2: Fix KMeans — drop students with incomplete records ──────
df_kmeans = df_wide.dropna(subset=['S1','S2','S3','S4','S5','S6','S7','S8'])
print(f"Students with full 8 sems: {len(df_kmeans)} out of {len(df_wide)}")

features = ['S1','S2','S3','S4','S5','S6','S7','S8']
X_cluster = df_kmeans[features].values

# Scale before clustering — important, SGPA range is small but still good practice
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X_cluster)

kmeans = KMeans(n_clusters=4, random_state=42, n_init=10)
df_kmeans = df_kmeans.copy()
df_kmeans['Cluster'] = kmeans.fit_predict(X_scaled)

# How do clusters map to actual Performance_Type?
print(df_kmeans.groupby(['Cluster', 'Performance_Type']).size().unstack(fill_value=0))
```

```python
# ── STEP 3: Predictive model — Sem 1-3 only → predict final SGPA ───
# Only students with complete data
df_model = df_wide.dropna(subset=['S1','S2','S3','S4','S5','S6','S7','S8'])

# Features: only early semester data (what a teacher knows mid-journey)
X = df_model[['S1', 'S2', 'S3']].values

# Target: average of final 3 sems (overall outcome)
df_model = df_model.copy()
df_model['Final_Avg'] = df_model[['S6', 'S7', 'S8']].mean(axis=1)
y = df_model['Final_Avg'].values

print(f"Training on {len(X)} students")
print(f"Features: Sem1, Sem2, Sem3 SGPA")
print(f"Target: avg of Sem6, Sem7, Sem8 SGPA\n")
```

```python
# ── STEP 4: Train and compare two models ───────────────────────────
models = {
    'Linear Regression': LinearRegression(),
    'Random Forest':     RandomForestRegressor(n_estimators=100, random_state=42)
}

for name, model in models.items():
    # Cross-validation (5-fold) — honest evaluation
    scores_r2  = cross_val_score(model, X, y, cv=5, scoring='r2')
    scores_mae = cross_val_score(model, X, y, cv=5, 
                                  scoring='neg_mean_absolute_error')
    
    print(f"{name}")
    print(f"  R²  : {scores_r2.mean():.3f}  (± {scores_r2.std():.3f})")
    print(f"  MAE : {(-scores_mae.mean()):.3f} SGPA points")
    print()
```

